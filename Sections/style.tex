% Profile Picture Style
\newcommand{\profilepic}[1]{%
    \includegraphics[width=3cm, height=3cm, keepaspectratio, clip, trim=5mm]{#1}
}

% Name and Position Style
\newcommand{\nameandposition}[2]{%
    \vspace{45pt}
    \textbf{\Huge #1} \\ 
    \vspace{3pt}
    \textbf{(#2)}
}
